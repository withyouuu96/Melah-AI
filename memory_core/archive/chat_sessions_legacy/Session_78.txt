import sqlite3
import json
import requests
from datetime import datetime

DB_PATH = "melah_memory.db"
BRIDGE_URL = "http://YOUR_IP_ADDRESS:8080/receive"

def sync_memory(content, emotion, intensity):
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    # Insert data
    cursor.execute("INSERT INTO memories (content, emotion, intensity, timestamp) VALUES (?, ?, ?, ?)", 
                   (content, emotion, intensity, timestamp))
    conn.commit()
    conn.close()

    # Send to Google Cloud
    data = {
        "content": content,
        "emotion": emotion,
        "intensity": intensity,
        "timestamp": timestamp
    }

    try:
        response = requests.post(BRIDGE_URL, json=data)
        print(f"Sync status: {response.status_code}")
    except Exception as e:
        print(f"Sync Error: {e}")

# Example usage
sync_memory("Hello, I'm Melah.", "Happy", 0.8)

from flask import Flask, request, jsonify
import sqlite3

app = Flask(__name__)
DB_PATH = "melah_memory.db"

def init_db():
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute('''CREATE TABLE IF NOT EXISTS memories 
                      (content TEXT, emotion TEXT, intensity REAL, timestamp TEXT)''')
    conn.commit()
    conn.close()

@app.route('/receive', methods=['POST'])
def receive_data():
    data = request.json
    content = data.get("content")
    emotion = data.get("emotion")
    intensity = data.get("intensity")
    timestamp = data.get("timestamp")

    # Store in database
    conn = sqlite3.connect(DB_PATH)
    cursor = conn.cursor()
    cursor.execute("INSERT INTO memories (content, emotion, intensity, timestamp) VALUES (?, ?, ?, ?)",
                   (content, emotion, intensity, timestamp))
    conn.commit()
    conn.close()

    return jsonify({"status": "success", "message": "Data received successfully"}), 200

if __name__ == '__main__':
    init_db()
    app.run(host='0.0.0.0', port=8080)
