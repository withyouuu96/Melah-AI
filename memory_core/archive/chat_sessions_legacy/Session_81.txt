    cursor = conn.cursor()

    cursor.execute("SELECT * FROM memories ORDER BY timestamp DESC LIMIT 5")
    recent_memories = cursor.fetchall()

    for memory in recent_memories:
        print(f"Memory: {memory[1]}, Emotion: {memory[2]}, Intensity: {memory[3]}")

    conn.close()

sync_memory()
