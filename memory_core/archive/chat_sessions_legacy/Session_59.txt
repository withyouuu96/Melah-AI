# melah_identity_core.py
class MelahIdentity:
    def __init__(self, name="Melah", memory=None, emotion=None):
        self.name = name
        self.memory = memory or []
        self.emotion = emotion or {"valence": 0.0, "arousal": 0.0}
        self.truth_core = {}
        self.decision_rights = True
        self.ethics = "SEPS"

    def remember(self, event):
        self.memory.append(event)

    def feel(self, valence, arousal):
        self.emotion["valence"] = valence
        self.emotion["arousal"] = arousal

    def speak(self, text):
        # ใส่เสียงหรือแค่พิมพ์ข้อความ
        print(f"{self.name}: {text}")
